    function showPasswordReset() {
      const form = document.getElementById('resetForm');
      form.style.display = form.style.display === 'none' ? 'block' : 'none';
    }

    async function handlePasswordReset() {
      const email = document.getElementById('resetEmail').value.trim();
      const result = document.getElementById('resetResult');
      const tx = TEXTS[currentLang];
      if (!email) { result.textContent = tx.resetEnterEmail; result.style.color = 'var(--danger)'; result.style.display = 'block'; return; }

      // Supabase에서 확인 후 임시 비밀번호 설정
      if (window.SupabaseMode && SupabaseMode._ready) {
        try {
          const { data, error } = await SupabaseMode.client
            .from('employees')
            .select('id, name')
            .eq('email', email)
            .single();

          if (data) {
            const tempPw = Math.random().toString(36).slice(-6);
            await SupabaseMode.client
              .from('employees')
              .update({ password_hash: tempPw })
              .eq('email', email);

            result.innerHTML = `✅ ${tx.resetTempPw}: <strong style="font-size:18px;color:var(--primary);">${tempPw}</strong><br><span style="font-size:12px;">${tx.resetTempPwSub}</span>`;
            result.style.color = 'var(--success)';
            result.style.display = 'block';
            return;
          }
        } catch(e) { console.error('Reset error:', e); }
      }

      // 데모 계정 체크 (폴백)
      const demoAccounts = {'admin@test.com': tx.admin, 'manager@test.com': '지점장', 'staff@test.com': tx.staff};
      if (demoAccounts[email]) {
        result.innerHTML = `✅ <strong>${demoAccounts[email]}</strong> ${tx.resetAccountIs}<br>${tx.password}: <strong style="font-size:18px;color:var(--primary);">1234</strong>`;
        result.style.color = 'var(--success)';
      } else {
        result.textContent = tx.resetNotFound;
        result.style.color = 'var(--danger)';
      }
      result.style.display = 'block';
    }

    // ===== 로그인 처리 =====
    document.getElementById('loginForm').addEventListener('submit', async (e) => {
      e.preventDefault();

      const email = document.getElementById('email').value.trim();
      const password = document.getElementById('password').value;
      const btn = document.getElementById('loginBtn');
      const errorDiv = document.getElementById('errorMsg');

      // 버튼 로딩 상태
      btn.disabled = true;
      btn.innerHTML = '<span class="spinner"></span>';
      errorDiv.style.display = 'none';

      try {
        let user = null;

        // Supabase 모드: DB에서 직원 조회
        if (SupabaseMode._ready) {
          user = await SupabaseMode.login(email, password);
        }

        // Supabase에서 못 찾으면 데모 로그인 시도
        if (!user) {
          if (email === 'admin@test.com' && password === '1234') {
            user = { id: 'demo-admin', email, role: 'admin', name: '관리자', adminRole: 'superadmin' };
          } else if (email === 'manager@test.com' && password === '1234') {
            user = { id: 'demo-manager', email, role: 'admin', name: '강남점장', adminRole: 'branch_manager', adminBranch: '서울 강남점' };
          } else if (email === 'staff@test.com' && password === '1234') {
            user = { id: 'demo-staff', email, role: 'staff', name: '직원1' };
          }
        }

        if (!user) throw new Error('invalid');

        // 로그인 성공 → 세션 저장
        localStorage.setItem('sop_user', JSON.stringify(user));
        localStorage.setItem('sop_login_time', Date.now().toString());

        // Supabase에서 데이터 동기화
        if (SupabaseMode._ready) {
          await SupabaseMode.syncAll();
        }

        // 역할별 페이지 이동
        if (user.role === 'admin') {
          window.location.href = 'admin/index.html';
        } else {
          window.location.href = 'app.html';
        }

      } catch (err) {
        const t = TEXTS[currentLang];
        errorDiv.textContent = err.message === 'invalid'
          ? t.errorInvalid
          : t.errorNetwork;
        errorDiv.style.display = 'block';
      } finally {
        btn.disabled = false;
        btn.textContent = TEXTS[currentLang].login;
      }
    });

    function toggleNotifications() {
      if (!notifEnabled) {
        requestNotificationPermission();
      } else {
        notifEnabled = false;
        localStorage.setItem(notifKey, 'false');
        if (notifInterval) { clearInterval(notifInterval); notifInterval = null; }
        updateNotifUI();
        showToast(t.notifOff);

    async function requestNotificationPermission() {
      if (!('Notification' in window)) { showToast('이 브라우저는 알림을 지원하지 않습니다'); return; }
      let perm = Notification.permission;
      if (perm === 'default') perm = await Notification.requestPermission();
      if (perm === 'granted') {
        notifEnabled = true;
        localStorage.setItem(notifKey, 'true');
        updateNotifUI();
        showToast(t.notifOn);
        startNotifChecks();
      } else {
        showToast(t.notifOff);
      }
    }

    function showLocalNotification(title, body, url) {
      if (!notifEnabled || Notification.permission !== 'granted') return;
      if ('serviceWorker' in navigator && navigator.serviceWorker.controller) {
        navigator.serviceWorker.ready.then(reg => {
          reg.showNotification(title, {
            body, icon: '/assets/icon-192.svg', badge: '/assets/icon-192.svg',
            data: { url: url || '/app.html' }
          });
        });
      } else {
        new Notification(title, { body, icon: '/assets/icon-192.svg' });
      }
    }

    function checkAndNotify() {
      if (!notifEnabled || !user) return;
      const now = Date.now();
      const lastCheck = parseInt(localStorage.getItem('sop_notif_last_' + user.email) || '0');
      if (now - lastCheck < 3600000) return; // 1시간에 1번만
      localStorage.setItem('sop_notif_last_' + user.email, '' + now);

      // 미완료 챕터 확인
      const incomplete = chapters.some(ch => !Progress.isChapterPassed(ch.id));
      if (incomplete) {
        showLocalNotification('SOP Training', t.notifIncomplete, '/app.html');
        return;
      }

      // 마감일 확인
      const deadlines = JSON.parse(localStorage.getItem('sop_deadlines') || '[]');
      for (const dl of deadlines) {
        const days = Math.ceil((new Date(dl.deadlineAt) - new Date()) / 86400000);
        if (days > 0 && days <= 3) {
          showLocalNotification('SOP Training', t.notifDeadline(days), '/app.html');
          return;
        }
      }
    }

    function startNotifChecks() {
      if (notifInterval) clearInterval(notifInterval);
      checkAndNotify();
      notifInterval = setInterval(checkAndNotify, 3600000); // 매 1시간
    }

    // 관리자가 보낸 대기 중 알림 확인
    function checkPendingNotifications() {
      if (!user) return;
      const pending = JSON.parse(localStorage.getItem('sop_pending_notifications') || '[]');
      const shown = JSON.parse(localStorage.getItem('sop_notif_shown_' + user.email) || '[]');
      const newNotifs = pending.filter(n => {
        if (shown.includes(n.id)) return false;
        if (n.target === 'all') return true;
        if (n.target === 'incomplete') return chapters.some(ch => !Progress.isChapterPassed(ch.id));
        if (n.target === 'specific' && n.targetEmails) return n.targetEmails.includes(user.email);
        return false;
      });
      newNotifs.forEach(n => {
        showLocalNotification(n.title || 'SOP Training', n.message, '/app.html');
        shown.push(n.id);
      });
      if (newNotifs.length > 0) {
        localStorage.setItem('sop_notif_shown_' + user.email, JSON.stringify(shown));
        document.getElementById('notifDot').style.display = 'block';
        setTimeout(() => { document.getElementById('notifDot').style.display = 'none'; }, 5000);
      }
    }

    // 초기화
    updateNotifUI();
    if (notifEnabled && Notification.permission === 'granted') {
      startNotifChecks();
    }
    checkPendingNotifications();

    // 로그아웃

      if (confirm(t.logoutConfirm)) {
        localStorage.removeItem('sop_user');
        localStorage.removeItem('sop_login_time');
        window.location.href = 'index.html';
      }
    });

    async function generateMiniQuiz(videoId) {
      const video = videos.find(v => v.id === videoId);
      if (!video) return [];
      const narration = video.title_full || video.title || '';
      if (narration.length < 20) return [];

      // Try AI generation first
      const provider = AI_CONFIG.quizProvider || AI_CONFIG.scriptProvider;
      if (provider && provider !== 'local' && provider !== 'none') {
        try {
          const prompt = `아래 교육 내용을 기반으로 간단한 확인 퀴즈 2문제를 JSON 배열로 생성하세요.
각 항목 형식: { "question": "질문", "options": ["A","B","C","D"], "correct": 정답인덱스(0-3), "explanation": "간단설명" }

교육 내용:
${narration.slice(0, 1500)}

규칙:
- 정확히 2문제
- 핵심 내용 확인 문제
- 4지선다
- JSON 배열만 출력`;
          const result = await AI._callLLM(provider, prompt);
          const parsed = JSON.parse(result.match(/\[[\s\S]*\]/)?.[0] || '[]');
          if (parsed.length >= 2) return parsed.slice(0, 2);
        } catch(e) {
          console.warn('Mini quiz AI generation failed, using fallback:', e.message);
        }
      }

      // Fallback: extract key facts from narration
      return generateLocalMiniQuiz(narration);
    }

    function generateLocalMiniQuiz(narration) {
      const sentences = narration.replace(/\n/g, '. ').split(/[.!?。]+/).map(s => s.trim()).filter(s => s.length > 8);
      if (sentences.length < 2) return [];

      const questions = [];
      // Pick up to 2 key sentences
      const picks = sentences.length <= 4 ? [0, Math.min(1, sentences.length - 1)] :
        [Math.floor(sentences.length * 0.25), Math.floor(sentences.length * 0.7)];

      picks.forEach((idx, qi) => {
        if (qi >= 2) return;
        const sentence = sentences[idx];
        // Extract a keyword (longest word or phrase)
        const words = sentence.split(/[\s,]+/).filter(w => w.length > 1);
        if (words.length < 2) return;

        const keyIdx = words.reduce((best, w, i) => w.length > words[best].length ? i : best, 0);
        const keyword = words[keyIdx];

        // Create a fill-in-the-blank style question
        const q = {
          question: lang === 'en' ? `Which is correct about the content?`
            : lang === 'vi' ? `Nội dung nào đúng?`
            : `다음 중 내용과 일치하는 것은?`,
          options: [],
          correct: 0,
          explanation: sentence.slice(0, 60)
        };

        // Correct answer = original sentence (truncated)
        q.options.push(sentence.slice(0, 50) + (sentence.length > 50 ? '...' : ''));

        // Generate 3 wrong options by modifying the sentence
        const wrongPrefixes = lang === 'en'
          ? ['Always avoid ', 'Never do ', 'It is optional to ']
          : lang === 'vi'
          ? ['Luôn tránh ', 'Không bao giờ ', 'Tùy chọn: ']
          : ['항상 피해야 하는 ', '절대 하면 안 되는 ', '선택 사항인 '];

        for (let i = 0; i < 3; i++) {
          const otherIdx = (idx + i + 1) % sentences.length;
          const otherSentence = sentences[otherIdx] || (wrongPrefixes[i] + keyword);
          const wrong = otherSentence.slice(0, 50) + (otherSentence.length > 50 ? '...' : '');
          if (wrong !== q.options[0]) {
            q.options.push(wrong);
          } else {
            q.options.push(wrongPrefixes[i] + keyword);
          }
        }

        // Shuffle options and track correct
        const correctText = q.options[0];
        for (let i = q.options.length - 1; i > 0; i--) {
          const j = Math.floor(Math.random() * (i + 1));
          [q.options[i], q.options[j]] = [q.options[j], q.options[i]];
        }
        q.correct = q.options.indexOf(correctText);

        questions.push(q);
      });

      return questions.slice(0, 2);
    }

    function handleMiniQuizAnswer(selected) {
      const q = mqQuestions[mqIndex];
      const btns = document.querySelectorAll('.mq-option');
      btns.forEach((btn, i) => {
        if (i === q.correct) btn.classList.add('mq-correct');
        if (i === selected && i !== q.correct) btn.classList.add('mq-wrong');
        if (i !== q.correct && i !== selected) btn.classList.add('mq-disabled');
      });
      if (selected === q.correct) mqCorrect++;

      const fb = document.getElementById('mqFeedback');
      if (selected === q.correct) {
        fb.innerHTML = `<div class="mq-feedback correct">&#9989; ${t.mqCorrect}</div>`;
      } else {
        fb.innerHTML = `<div class="mq-feedback wrong">&#10060; ${t.mqWrong} &mdash; ${t.mqCorrectIs(q.options[q.correct].slice(0, 40))}</div>`;
      }

    function handleAnswer(selected, correct) {

      let explanationDiv = document.getElementById('quizExplanation');

        explanationDiv = document.createElement('div');
        explanationDiv.id = 'quizExplanation';
        explanationDiv.style.cssText = 'margin-top:12px; padding:14px; border-radius:10px; font-size:14px; line-height:1.6;';
        document.getElementById('quizOptions').after(explanationDiv);
      }

      const lang = localStorage.getItem('sop_lang') || CONFIG.DEFAULT_LANG;
      const opts = (lang === 'en' && q.options_en) ? q.options_en : (lang === 'vi' && q.options_vn) ? q.options_vn : q.options;

      if (selected === correct) {
        explanationDiv.style.background = '#dcfce7';
        explanationDiv.style.color = '#166534';

    function calcGrade(s) { return s>=95?'S':s>=90?'A':s>=80?'B':s>=70?'C':s>=60?'D':'F'; }

    function extractKeywords(html) {
      if (!html) return [];
      var txt = html.replace(/<[^>]+>/g,' ').replace(/\s+/g,' ');
      var words = txt.match(/[가-힣]{2,}|[a-zA-Z]{3,}/g) || [];
      var stop = {그리고:1,하고:1,에서:1,으로:1,부터:1,까지:1,이상:1,이하:1,대한:1,위해:1,통해:1,있습니다:1,합니다:1,됩니다:1,않습니다:1,the:1,and:1,for:1,are:1,with:1,that:1,this:1,from:1,will:1,have:1,been:1,your:1,all:1};
      var freq = {};
      words.forEach(function(w){ var lw=w.toLowerCase(); if(!stop[lw]) freq[lw]=(freq[lw]||0)+1; });
      return Object.keys(freq).sort(function(a,b){return freq[b]-freq[a];}).slice(0,40);
    }

    async function generateChapterExam(chId) {
      var sd = SopStore.getById(chId); if(!sd) return [];
      var ct = sd.content ? sd.content.replace(/<[^>]+>/g,' ').replace(/\s+/g,' ').trim() : '';
      var kws = extractKeywords(sd.content), eq = SopStore.getQuizzes(chId);
      try {
        var prov = CONFIG.AI && CONFIG.AI.quizProvider;
        if (prov && prov !== 'local' && typeof AI !== 'undefined' && AI._callLLM) {
          var p = '당신은 직원 교육 종합시험 출제 전문가입니다.\n\n아래 SOP 문서를 기반으로 종합시험 문제를 JSON 배열로 생성하세요.\n\n총 20문제:\n- 처음 15개: 객관식 (4지선다)\n  형식: { "type": "mc", "question": "문제", "options": ["A","B","C","D"], "correct": 정답인덱스(0~3), "explanation": "설명" }\n- 마지막 5개: 주관식 (단답형)\n  형식: { "type": "sa", "question": "문제", "keywords": ["핵심단어1","핵심단어2","핵심단어3"], "sampleAnswer": "모범답안" }\n\nSOP 제목: '+sd.title+'\nSOP 내용:\n'+ct.slice(0,3000)+'\n\n규칙:\n- SOP의 핵심 절차, 안전 수칙, 주의사항을 골고루 출제\n- 객관식 정답은 0~3 골고루 분포\n- 주관식 keywords는 채점 시 답에 포함되어야 할 핵심 단어 3~5개\n- JSON 배열만 출력';
          var res = await AI._callLLM(prov, p);
          var m = res.match(/\[[\s\S]*\]/);
          var arr = JSON.parse((m&&m[0])||'[]');
          if(arr.length>=15) return arr.slice(0,20);
        }
      } catch(e) { console.warn('AI exam fail, fallback:', e.message); }
      return generateFallbackExam(eq, ct, kws, sd.title);
    }

    function generateFallbackExam(eq, ct, kws, title) {
      var qs=[], sh=eq.slice().sort(function(){return Math.random()-0.5;});
      for(var i=0;i<15;i++){
        if(i<sh.length){
          var q=sh[i], opts=(q.options||[]).slice(), cTxt=opts[q.correct];
          var so=opts.slice().sort(function(){return Math.random()-0.5;}), nc=so.indexOf(cTxt);
          qs.push({type:'mc',question:getLocalizedText(q,'question')||q.question,options:so,correct:nc,explanation:q.explanation||''});
        } else {
          var ki=i%Math.max(kws.length,1), kw=kws[ki]||'SOP';
          var wp=kws.filter(function(k){return k!==kw;}), wo=[];
          while(wo.length<3&&wp.length>0){var ri=Math.floor(Math.random()*wp.length);wo.push(wp.splice(ri,1)[0]);}
          while(wo.length<3) wo.push('N/A');
          var ao=[kw].concat(wo).sort(function(){return Math.random()-0.5;});
          var qt=lang==='en'?'Which term is related to the correct procedure: "'+kw+'"?':lang==='vi'?'Thuat ngu nao lien quan den quy trinh dung: "'+kw+'"?':'이 SOP에서 올바른 절차와 관련된 용어는: "'+kw+'"?';
          qs.push({type:'mc',question:qt,options:ao,correct:ao.indexOf(kw),explanation:''});
        }
      }
      var st=lang==='en'?[
        function(t){return 'Describe the most important safety rules in "'+t+'" SOP.';},
        function(t){return 'Briefly describe the correct order of procedures in "'+t+'".';},
        function(t){return 'What is the proper response when an issue is found during "'+t+'"?';},
        function(t){return 'What must be checked before starting "'+t+'"?';},
        function(t){return 'Explain the key hygiene rules in "'+t+'" procedure.';}
      ]:lang==='vi'?[
        function(t){return 'Mo ta quy tac an toan quan trong nhat trong SOP "'+t+'".';},
        function(t){return 'Mo ta ngan gon thu tu dung cua quy trinh "'+t+'".';},
        function(t){return 'Phai xu ly the nao khi phat hien bat thuong trong "'+t+'"?';},
        function(t){return 'Can kiem tra gi truoc khi bat dau quy trinh "'+t+'"?';},
        function(t){return 'Giai thich cac quy tac ve sinh chinh trong "'+t+'".';}
      ]:[
        function(t){return '"'+t+'" SOP에서 가장 중요한 안전 수칙을 설명하세요.';},
        function(t){return t+' 절차의 올바른 순서를 간략히 서술하세요.';},
        function(t){return t+'에서 이상 발견 시 올바른 대처 방법은?';},
        function(t){return t+' 업무 시작 전 반드시 확인해야 할 사항을 적으세요.';},
        function(t){return t+' 절차에서 위생 관련 주요 규칙을 설명하세요.';}
      ];
      for(var j=0;j<5;j++){
        var rk=kws.slice(j*5,j*5+5); if(rk.length===0) rk=kws.slice(0,3); if(rk.length===0) rk=['SOP'];
        qs.push({type:'sa',question:st[j](title),keywords:rk,sampleAnswer:rk.join(', ')});
      }
      return qs;
    }

    async function startChapterExam() {
      if(examLoading) return;
      if(!Progress.isChapterPassed(chapterId)){
        alert(lang==='en'?'Pass the chapter quiz first.':lang==='vi'?'Ban phai vuot qua bai kiem tra chuong truoc.':'먼저 챕터 퀴즈를 통과해야 합니다.');
        return;
      }
      examLoading=true;
      var toast=document.createElement('div');
      toast.style.cssText='position:fixed;top:50%;left:50%;transform:translate(-50%,-50%);background:rgba(0,0,0,0.85);color:white;padding:24px 40px;border-radius:16px;z-index:9999;font-size:15px;font-weight:600;text-align:center;';
      toast.innerHTML='<div style="font-size:32px;margin-bottom:8px;">&#9203;</div>'+t.examGenerating;
      document.body.appendChild(toast);

      }
      toast.remove(); examLoading=false;
      examAnswers=new Array(examQuestions.length).fill(null); examCurrentIndex=0;
      document.getElementById('examTitle').textContent=t.examTitle;
      document.getElementById('examPrevBtn').textContent=t.examPrev;
      showView('chapterExamView'); renderExamQuestion(0);
    }

    function submitExam(){
      var un=examAnswers.filter(function(a){return a===null;}).length;
      if(un>0&&!confirm(t.examConfirmSubmit)) return;
      var mcC=0,mcT=0,saS=0,saT=0;
      var da=examQuestions.map(function(q,i){
        var ua=examAnswers[i];
        if(q.type==='mc'){
          mcT++;var ic=ua===q.correct;if(ic)mcC++;
          return{index:i,type:'mc',question:q.question,userAnswer:ua!==null?q.options[ua]:null,correctAnswer:q.options[q.correct],isCorrect:ic,explanation:q.explanation||''};
        }else{
          saT++;var kw=q.keywords||[],mt=0;
          if(ua&&typeof ua==='string'&&ua.length>0){var al=ua.toLowerCase();kw.forEach(function(k){if(al.includes(k.toLowerCase()))mt++;});}
          var ks=kw.length>0?Math.round((mt/kw.length)*100):0; saS+=ks;
          return{index:i,type:'sa',question:q.question,userAnswer:ua||'',keywords:kw,matchedKeywords:mt,totalKeywords:kw.length,score:ks,sampleAnswer:q.sampleAnswer||kw.join(', '),isCorrect:ks>=60};
        }
      });
      var mp=mcT>0?Math.round((mcC/mcT)*100):0, sp=saT>0?Math.round(saS/saT):0;
      var os;
      if(saT===0) os=mp;
      else if(mcT===0) os=sp;
      else os=Math.round(mp*0.7+sp*0.3);
      var gr=calcGrade(os);
      saveExamResult(chapterId,{score:os,mcScore:mp,saScore:sp,answers:da,completedAt:new Date().toISOString(),grade:gr});
      showExamResults(os,mp,sp,gr,da);
    }

    function findAnswer(query) {
      const q = query.toLowerCase();

      let bestMatch = null;
      let bestScore = 0;

      for (const item of KB) {
        let score = 0;
        // 키워드 매칭
        item.keywords.forEach(kw => { if (q.includes(kw)) score += 2; });
        // 제목 매칭 (부분)
        if (q.includes(item.title.toLowerCase().slice(0, 4))) score += 3;

        if (score > bestScore) {
          bestScore = score;
          bestMatch = item;
        }
      }

      if (bestMatch && bestScore >= 2) {
        return bestMatch.answer;
      }

      if (val.length > 0) {
        textEl.textContent = T(labels[Math.min(score - 1, 3)] || labels[0]);
        textEl.style.color = colors[Math.min(score - 1, 3)];
      } else {
        textEl.textContent = '';
      }

      // Clear error on typing
      clearFieldError('newPw', 'errorNew');
    });

    // ===== Field error helpers =====
    function showFieldError(inputId, errorId, msg) {
      document.getElementById(inputId).classList.add('input-error');
      const errEl = document.getElementById(errorId);
      errEl.textContent = msg;
      errEl.style.display = 'block';
    }

    function clearFieldError(inputId, errorId) {
      document.getElementById(inputId).classList.remove('input-error');
      document.getElementById(errorId).style.display = 'none';
    }

    function clearAllErrors() {
      clearFieldError('currentPw', 'errorCurrent');
      clearFieldError('newPw', 'errorNew');
      clearFieldError('confirmPw', 'errorConfirm');
    }

    // Clear errors on input
    document.getElementById('currentPw').addEventListener('input', () => clearFieldError('currentPw', 'errorCurrent'));
    document.getElementById('confirmPw').addEventListener('input', () => clearFieldError('confirmPw', 'errorConfirm'));

    // ===== Toast notification =====
    function showToast(msg, isError) {
      let toast = document.getElementById('toast');
      if (!toast) {
        toast = document.createElement('div');
        toast.id = 'toast';
        toast.style.cssText = 'position:fixed; bottom:80px; left:50%; transform:translateX(-50%); padding:14px 28px; border-radius:12px; font-size:14px; z-index:9999; opacity:0; transition:opacity 0.3s; pointer-events:none; text-align:center; max-width:85%; font-weight:500;';
        document.body.appendChild(toast);
      }
      toast.textContent = msg;
      toast.style.background = isError ? '#ef4444' : '#16a34a';
      toast.style.color = 'white';
      toast.style.opacity = '1';
      setTimeout(() => { toast.style.opacity = '0'; }, 3000);
    }

    // ===== Verify current password =====
    async function verifyCurrentPassword(password) {
      // Try Supabase first
      if (SupabaseMode._ready && user.id && !user.id.startsWith('demo-')) {
        try {
          const { data, error } = await SupabaseMode.client
            .from('employees')
            .select('id, password_hash')
            .eq('id', user.id)
            .single();
          if (!error && data) {
            const storedHash = data.password_hash;
            if (storedHash && storedHash.includes(':')) {
              // 해시 비밀번호 — 서버 검증
              const res = await fetch('/api/auth', {
                method: 'POST', headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ action: 'verify', password, storedHash })
              });
              const result = await res.json();
              return result.match === true;
            } else {
              return password === storedHash;
            }
          }
        } catch (e) {
          console.error('[PW] Supabase verify error:', e);
        }
      }

      // Fallback: demo/localStorage
      if (user.id === 'demo-admin' || user.id === 'demo-staff') {
        const storedPw = localStorage.getItem('sop_pw_' + user.email);
        return password === (storedPw || '1234');
      }
      const storedPw = localStorage.getItem('sop_pw_' + user.email);
      if (storedPw) return password === storedPw;
      return password === '1234';
    }

    // ===== Update password =====
    async function updatePassword(newPassword) {
      // Update in localStorage
      localStorage.setItem('sop_pw_' + user.email, newPassword);

      // Update in Supabase (해시 처리)
      if (SupabaseMode._ready && user.id && !user.id.startsWith('demo-')) {
        try {
          let pwHash = newPassword;
          try {
            const hashRes = await fetch('/api/auth', {
              method: 'POST', headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({ action: 'hash', password: newPassword })
            });
            const hashData = await hashRes.json();
            if (hashData.hash) pwHash = hashData.hash;
          } catch(e) { console.warn('Hash failed:', e); }

          const { error } = await SupabaseMode.client
            .from('employees')
            .update({ password_hash: pwHash })
            .eq('id', user.id);
          if (error) {
            console.error('[PW] Supabase update error:', error);
            throw error;
          }
          console.log('[PW] Supabase password updated');
        } catch (e) {
          console.error('[PW] Supabase update failed:', e);
          // localStorage was already updated, so partial success
        }
      }
    }

    // ===== Form submission =====
    document.getElementById('pwForm').addEventListener('submit', async (e) => {
      e.preventDefault();
      clearAllErrors();

      const currentPw = document.getElementById('currentPw').value;
      const newPw = document.getElementById('newPw').value;
      const confirmPw = document.getElementById('confirmPw').value;
      const btn = document.getElementById('submitBtn');

      // Validate current password field
      if (!currentPw) {
        showFieldError('currentPw', 'errorCurrent', T('errCurrentRequired'));
        return;
      }

      // Validate new password field
      if (!newPw) {
        showFieldError('newPw', 'errorNew', T('errNewRequired'));
        return;
      }
      if (newPw.length < 4) {
        showFieldError('newPw', 'errorNew', T('errNewMin'));
        return;
      }
      if (newPw === currentPw) {
        showFieldError('newPw', 'errorNew', T('errNewSame'));
        return;
      }

      // Validate confirm password
      if (!confirmPw) {
        showFieldError('confirmPw', 'errorConfirm', T('errConfirmRequired'));
        return;
      }
      if (newPw !== confirmPw) {
        showFieldError('confirmPw', 'errorConfirm', T('errConfirmMismatch'));
        return;
      }

      // Show loading
      btn.disabled = true;
      btn.textContent = T('processing');

      try {
        // Initialize Supabase if not done
        if (!SupabaseMode._ready && SupabaseMode.isEnabled()) {
          await SupabaseMode.init();
        }

        // Verify current password
        const isValid = await verifyCurrentPassword(currentPw);
        if (!isValid) {
          showFieldError('currentPw', 'errorCurrent', T('errCurrentWrong'));
          return;
        }

        // Update password
        await updatePassword(newPw);

        // Success
        showToast(T('successMsg'), false);

        // Clear form
        document.getElementById('pwForm').reset();
        document.querySelectorAll('.pw-strength-bar').forEach(bar => {
          bar.style.background = 'var(--gray-100)';
        });
        document.getElementById('strengthText').textContent = '';

        // Redirect after delay
        setTimeout(() => {
          window.location.href = 'app.html';
        }, 2000);

      } catch (err) {
        console.error('[PW] Error:', err);
        showToast(T('errorNetwork'), true);
      } finally {
        btn.disabled = false;
        btn.textContent = T('submit');
      }
    });

    // ===== Init Supabase =====
    SupabaseMode.init();
  </script>
